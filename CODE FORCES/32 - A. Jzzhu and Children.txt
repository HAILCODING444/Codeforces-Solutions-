#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t ;
    cin>>t ; 
    int a[t] ;
    double  m ; 
    cin>>m ; 
    int k=0 ; 
    while(t--){
        cin>>a[k] ; 
        if(a[k] <= m){
            a[k] = 0 ; 
        }
        k++ ; 
    }
    int maxo = 0 , ans = 0 ; 
    
    for(int i=0 ; i<k ; i++){
        if(ceil(a[i]/m) >= maxo){
            maxo = ceil(a[i]/m) ; 
            ans = i ; 
        }
    }
    
    cout<<ans+1 ; 

    return 0;
}
