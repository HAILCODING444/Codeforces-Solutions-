

#include <bits/stdc++.h>

using namespace std;

void recurrsiveFUNCTN(vector<int> &ans , int x){
    if(x == 0 ){
        return ; 
    }
    
    int temp = ans[x] ; 
    ans[x] = ans[x-1] ; 
    ans[x-1] = temp ; 
    
    recurrsiveFUNCTN(ans , x-1 ) ; 
}

int main()
{
    int n ; 
    cin>>n ;
    int size = n ;
    if(size == 1){
        cout<<1 ; 
    }else{
        vector <int> ans ;
        int i=1 ; 
        while(n--){
            ans.push_back(i++); 
            
        }
        recurrsiveFUNCTN(ans, size-1);
        for(int i=0 ; i<ans.size() ; i++){
            cout<<ans[i]<<" " ;
            
        }
    }
    return 0;
}
