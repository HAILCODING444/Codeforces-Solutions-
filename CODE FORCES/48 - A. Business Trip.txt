
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n ; 
    cin>>n ; 
    int t=12 ; 
    vector<int> a ; 
    while(t--){
        int x ; 
        cin>>x ; 
        a.push_back(x) ; 
    }
    
    sort(a.begin() , a.end()) ; 
    int count =0 ; 
    for(int i=a.size()-1 ; i>=0 ; i--){
        if(n <= 0) break ; 
        n -= a[i] ; 
        count++ ; 
    }
    
    if(n > 0) cout<<-1 ; 
    else cout<<count;

    return 0;
}
