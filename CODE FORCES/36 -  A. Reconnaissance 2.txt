#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n ; 
    cin>>n ; 
    int a[n] ;
    
    for(int i=0 ; i<n ; i++){
        cin>>a[i] ; 
    }
    
    int min = INT_MAX ;
    int index1 =0 ; 
    int index2 =0 ; 
    int diff = abs(a[0]-a[n-1]) ; 
    if(diff < min ){
        min = diff ; 
        index1 =0 ; 
        index2 = n-1 ; 
    }
    
    for(int i=1 ; i<n ; i++){
        diff = abs(a[i-1] - a[i]) ; 
        if(diff < min ){
            min = diff ; 
            index1 =i-1 ; 
            index2 =i; 
             
        }
    }
    
    cout<<index1+1<<" "<<index2+1 ; 
    
    return 0;
}
