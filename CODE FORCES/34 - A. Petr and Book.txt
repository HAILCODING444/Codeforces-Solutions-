
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int a ;
    cin>>a ;
    int b[7] ; 
    for(int i=0 ; i<7 ; i++){
        cin>>b[i] ; 
    }
    int i=0 ; 
    while(true){
        a = a - b[i] ;
        if(a <= 0){
            break ; 
        }
        i++ ;
        if(i > 6){
            i=0 ; 
        }
    }
    cout<<i+1 ; 
    
    return 0;
}
