#include <bits/stdc++.h>

using namespace std;

int main()
{
    int s , n ; 
    cin>>s>>n ;
    vector<pair<int , int>> ans ; 
    while(n--){
        int x , y ; 
        cin>>x>>y ; 
        ans.push_back({x , y}) ; 
    }
    
    sort(ans.begin() , ans.end()) ;
    
    bool flag = 1 ; 
    for(int i=0 ; i<ans.size() ; i++){
        if(s > ans[i].first){
            s += ans[i].second ; 
        }else{
            flag = 0 ; 
            break ; 
        }
    }
    
    // for(int i=0 ; i<ans.size() ; i++){
    //     cout<<ans[i].first<<" "<<ans[i].second<<endl ; 
    // }
    
    if(flag) cout<<"YES";
    else cout<<"NO" ; 

    return 0;
}
