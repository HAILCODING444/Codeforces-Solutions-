#include <cstring>
using namespace std;
bool checkAB(char input[]) {
	// Write your code here
    if(strlen(input) == 0){
        return true;
    }
    if(input[0] != 'a'){
        return false;
    }
    if(input[0] == 'a' && input[1] == '\0'){
        return true;
    }
    if(input[0] == 'a' && input[1] == 'a'){
        return checkAB(input + 1);
    }
    if(input[0] == 'a' && input[1] == 'b' && input[2] == 'b'){
        return checkAB(input + 3);
    }
    else{
        return false;
    }
}
