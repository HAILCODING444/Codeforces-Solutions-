#include <bits/stdc++.h>

using namespace std;

int main()
{
    string a ; 
    cin>>a ; 
    int n = a.size() ; 
    int i=0 ; 
    int count1 =0 ;
    int count0 =0 ;
    bool flag =0 ; 
    while(i < n){
        if(a[i] == '1'){
            count0 = 0 ; 
            count1++ ; 
        }else if(a[i] == '0'){
            count1=0 ; 
            count0++ ; 
        }
        if(count1 == 7 || count0 == 7){
            cout<<"YES" ;
            flag = 1 ; 
            break ; 
        }
        i++ ; 
    }
    
    if(!flag) cout<<"NO" ; 
    return 0;
}
