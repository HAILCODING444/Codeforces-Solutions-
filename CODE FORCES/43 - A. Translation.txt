#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    string a  , b;
    cin>>a>>b ;
    bool flag = 0 ; 
    if(a.length() != b.length()){
        cout<<"NO" ; 
    }else{
        int i=0 ; 
        int j= b.length()-1 ; 
        while(i<a.length()){
            if(a[i] == b[j]){
                i++;
                j--;
            }else{
                flag=1 ; 
                break ; 
            }
        }
        if(flag) cout<<"NO" ; 
        else cout<<"YES" ; 
    }
 
    return 0;
}
