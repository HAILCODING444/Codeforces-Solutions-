#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    vector<int> a(25 , false) ; 
    string b  ; 
    cin>>b ; 
    int n = b.size() ;
    int count = 0 ; 
    for(int i=0 ; i<n ; i++){
        if(!a[b[i]-'a']){
           a[b[i]-'a'] = true ;
           count++ ; 
        }
    }
    if(count %2 == 0){
        cout<<"CHAT WITH HER!" ; 
    }else{
        cout<<"IGNORE HIM!" ; 
    }
    return 0;
}
