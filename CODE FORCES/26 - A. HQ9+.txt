#include <bits/stdc++.h>

using namespace std;

int main()
{
    string a ; 
    cin>>a ; 
    
    int n = a.size() ; 
    bool flag = 0 ; 
    for(int i=0 ; i<n ; i++){
        if(a[i] == 'H' || a[i]=='Q' || a[i]=='9'){
            flag = 1 ; 
            cout<<"YES" ;
            break ; 
        } 
    }
    
    if(flag == 0) cout<<"NO" ; 

    return 0;
}
